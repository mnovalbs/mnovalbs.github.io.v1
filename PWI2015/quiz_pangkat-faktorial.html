Pangkat

^

Hasil : 

<form>
  <input id='bilanganA' type='number' min='0' value='1'/> ^ <input id='bilanganB' type='number' min='1' value='1'/>
  <button type='button' onclick='pangkat()'>Hitung</button>
  <p>Hasil : <span id='hasilPangkat'></span></p>
</form>
<script>
  function pangkat(){
    var bilanganA = document.getElementById("bilanganA").value;
    var bilanganB = document.getElementById("bilanganB").value;
    var hasil = 1;
    for(i=1;i<=bilanganB;i++){
      hasil *= bilanganA;
    }
    document.getElementById("hasilPangkat").innerHTML = hasil;
  }
</script>

Faktorial

Hasil : 

<form>
  <input id='faktorA' type='number' min='0' value='1'/>
  <button type='button' onclick='faktorial()'>Hitung</button>
  <p>Hasil : <span id='hasilFaktorial'></span></p>
</form>
<script>
  function faktorial(){
    var faktorA = document.getElementById("faktorA").value;
    var hasilFaktorial = recursiveFaktorial(faktorA);

    document.getElementById("hasilFaktorial").innerHTML = hasilFaktorial;
  }

  function recursiveFaktorial(bilangan){
    if(bilangan<2){
      return 1;
    }else{
      return bilangan * recursiveFaktorial(bilangan-1);
    }
  }
</script>
