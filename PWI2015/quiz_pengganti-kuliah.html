Javascript Object

function Product(nama,harga){
  this.nama   = nama;
  this.harga  = harga;
  var _diskon;
  Object.defineProperty(this,"diskon",{ //Mendefinisikan this.diskon
    get: function(){
      return _diskon; //Nilai kembalian apabila dipanggil NamaObject.diskon
    },
    set: function(value){ //Set Nilai apabila dipanggil NamaObject.diskon = isi
      _diskon = value;
      if(_diskon>25){
        _diskon = 25;
      }
    }
  });
}

//script pada klien
var kaos = new Product("Pollo",100000);
kaos.diskon = 20; //potongan pertama
kaos.diskon += 20; //potongan kedua
kaos.harga -= kaos.harga * (kaos.diskon/100);
document.write("Harga "+kaos.nama+" setelah diskon = "+kaos.harga);
